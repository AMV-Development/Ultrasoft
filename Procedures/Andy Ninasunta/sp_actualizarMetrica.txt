ALTER PROCEDURE [dbo].[sp_actualizarMetrica]   
  @idMetrica int,   
    @nombreMetrica nvarchar(30),   
    @simbolo char(3)   
AS   

UPDATE METRICA 
SET NOMBRE_METRICA=@nombreMetrica,
	SIMBOLO=@simbolo
WHERE ID_METRICA=@idMetrica


GO
