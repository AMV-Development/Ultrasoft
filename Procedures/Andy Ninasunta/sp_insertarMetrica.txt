ALTER PROCEDURE [dbo].[sp_insertarMetrica]   
    @nombreMetrica nvarchar(30),   
    @simbolo char(3)   
AS   
DECLARE @id int

SELECT @id = MAX(ID_METRICA) + 1 FROM METRICA WITH (UPDLOCK, HOLDLOCK)


    INSERT INTO METRICA (ID_METRICA, NOMBRE_METRICA, SIMBOLO)
VALUES (@id, @nombreMetrica, @simbolo);


GO
